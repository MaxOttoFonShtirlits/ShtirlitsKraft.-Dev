ability = {
    # 1. Отправить генерала США в Англию
    send_general_to_eng = {
        name = ABILITY_SEND_GENERALS
        desc = ABILITY_SEND_GENERALS_DESC
        sound_effect = command_power_ability_offensive
        type = army_leader

        allowed = {
            is_border_war = no
            OWNER = { tag = USA }
            is_in_faction_with = ENG
        }

        ai_will_do = { factor = -1 }

        cost = 0
        duration = 0
        fixed_random_seed = no

        one_time_effect = {
            if = {
                limit = { is_in_faction_with = ENG }

                    add_trait = { trait = us_foreign_general }
                    set_nationality = ENG
                }
            }
            else = {
                ROOT = { retire = yes }
            }
        }
    }
