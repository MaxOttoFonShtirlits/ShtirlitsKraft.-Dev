has_green_air = {
	has_country_flag = FAI_green_air
}

has_red_air = {
	NOT = { has_country_flag = FAI_green_air }
}

has_manpower_and_equipment_surplus = {
	manpower_per_military_factory > 1000
	check_variable = { FAI_fielded_eq_ratio > 0.90 }
	NOT = { has_country_flag = FAI_eq_crisis }
	NOT = { has_country_flag = FAI_manpower_crisis }
}

has_manpower_or_equipment_shortage = {
	OR = {
		has_country_flag = FAI_eq_crisis
		has_country_flag = FAI_manpower_crisis
		manpower_per_military_factory < 500
		check_variable = { FAI_fielded_eq_ratio < 0.70 }
	}
}

has_an_actual_civil_war = {
	has_civil_war = yes
	NOT = {
		any_enemy_country = {
			has_civil_war = no
		}
	}
}

FAI_mobile_doctrine = {
	has_doctrine = new_mobile_warfare
}

FAI_superior_firepower_doctrine = {
	has_doctrine = superior_firepower
}

FAI_grand_battleplan_doctrine = {
	has_doctrine = grand_battleplan
}

FAI_mass_assault_doctrine = {
	has_doctrine = mass_assault
}

FAI_known_vanilla_grand_doctrine = {
	OR = {
		FAI_mobile_doctrine = yes
		FAI_superior_firepower_doctrine = yes
		FAI_grand_battleplan_doctrine = yes
		FAI_mass_assault_doctrine = yes
	}
}

FAI_unknown_or_modded_doctrine = {
	NOT = { FAI_known_vanilla_grand_doctrine = yes }
}

FAI_no_crisis_hold = {
	NOT = { has_country_flag = FAI_eq_crisis }
	NOT = { has_country_flag = FAI_manpower_crisis }
}
