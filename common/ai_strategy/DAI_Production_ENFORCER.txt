# NOTE ON has_equipment TRIGGERS:
# has_equipment cannot check negative values - it clamps at 0 and returns false.
# All "deficit" checks below use low positive thresholds instead:
#   < 50  = dangerously low / near-empty stockpile  -> moderate response
#   < 1   = literally zero or trace amounts         -> emergency response
# These two layers stack additively for a proportional escalation.

# MECHANIZED TRANSITION BRIDGE
# When the AI just researched mechanized infantry it wants mech variants for tank templates
# but mech production hasn't ramped yet - this prevents the armor production collapse
# during that window by keeping motorized alive and hammering mech production hard.
# Aborts once mech stockpile hits 500 - at that point the dedicated SP strategies take over.
mechanized_transition_bridge = {
	enable = {
		has_tech = mechanised_infantry
		has_equipment = { mechanized_equipment < 0 }
	}
	abort_when_not_enabled = yes

	# Keep producing motorized - the AI has motorized stock that can still support tank divisions
	# Don't let it crater just because mech exists now
	#ai_strategy = {
	#	type = equipment_production_factor
	#	id = motorized
	#	value = 60
	#}

	# Slam mech production to ramp up as fast as possible
	ai_strategy = {
		type = equipment_production_factor
		id = mechanized
		value = 2000
	}

	ai_strategy = {
		type = equipment_variant_production_factor
		id = mechanized_equipment
		value = 3000
	}

	# Don't dump motorized divisions either - they're still useful while mech ramps
	ai_strategy = {
		type = equipment_production_min_factories_archetype
		id = motorized_equipment
		value = 2
	}
}
