# FAI generic naval invasion manager
# Compatibility rule: no tag gates, date gates, state IDs, focus checks, or vanilla-only target assumptions.

FAI_invasion_target_not_valid_anymore = {
	enable = {
		has_war = yes
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			OR = {
				controller = {
					NOT = { has_war_with = FROM.FROM }
				}
				owner = {
					NOT = { has_war_with = FROM.FROM }
				}
			}
		}
		value = -100
	}
}

FAI_invasion_prefer_ports_and_real_targets = {
	enable = {
		has_war = yes
		FAI_no_crisis_hold = yes
	}

	abort_when_not_enabled = yes

	# Ports are the main compatible target. They create supply and stop landings from becoming trapped beach raids.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			any_province_building_level = {
				province = { all_provinces = yes }
				building = naval_base
				level > 0
			}
		}
		value = 35
	}

	# High-value enemy-controlled states stay attractive even in overhaul maps.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			check_variable = { state_strategic_value > 15 }
		}
		value = 20
	}

	# Enemy cores usually matter more than random occupied coastline.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			is_core_of = FROM
		}
		value = 12
	}

	# Port targets next to our land are excellent follow-up targets: short path, immediate front pressure.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			any_neighbor_state = {
				is_controlled_by = FROM.FROM
			}
			any_province_building_level = {
				province = { all_provinces = yes }
				building = naval_base
				level > 0
			}
		}
		value = 25
	}

	# Random non-port beaches are usually how the AI throws divisions into supply death.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			NOT = {
				any_province_building_level = {
					province = { all_provinces = yes }
					building = naval_base
					level > 0
				}
			}
		}
		value = -30
	}

	# Low-value non-port beaches get buried hard unless another strategy has a very strong reason.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			check_variable = { state_strategic_value < 8 }
			NOT = {
				any_province_building_level = {
					province = { all_provinces = yes }
					building = naval_base
					level > 0
				}
			}
		}
		value = -60
	}
}


FAI_invasion_foothold_followup = {
	enable = {
		has_war = yes
		FAI_no_crisis_hold = yes
		alliance_strength_ratio > 0.75
	}

	abort_when_not_enabled = yes

	# Once a port is taken, feed the beachhead enough units to become an actual front.
	ai_strategy = {
		type = front_unit_request
		state_trigger = {
			is_controlled_by = FROM.FROM
			all_neighbor_state = {
				controller = {
					has_war_with = FROM.FROM
				}
			}
			any_province_building_level = {
				province = { all_provinces = yes }
				building = naval_base
				level > 0
			}
		}
		value = 45
	}

	# Push out from ports before the enemy seals the landing.
	ai_strategy = {
		type = front_control
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			all_neighbor_state = {
				is_controlled_by = FROM.FROM
				any_province_building_level = {
					province = { all_provinces = yes }
					building = naval_base
					level > 0
				}
			}
		}
		ratio = 0.0
		priority = 7200
		ordertype = front
		execution_type = rush_weak
		##execute_order = yes
		manual_attack = yes
	}
}
