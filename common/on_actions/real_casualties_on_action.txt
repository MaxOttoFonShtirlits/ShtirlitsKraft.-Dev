on_actions = {
    on_monthly = {
        effect = {
            if = {
                limit = {
                    check_variable = { casualties_calculated_deaths_k > casualties_k }
                }
                set_variable = { casualties_calculated_deaths_k = casualties_k }
            }
            if = {
                limit = {
                    check_variable = { casualties_k > casualties_calculated_deaths_k }
                    check_variable = { max_manpower_k > 0 }
                }
                set_temp_variable = { temp_delta_k = casualties_k }
                subtract_from_variable = { temp_delta_k = casualties_calculated_deaths_k }
                every_controlled_state = {
                    limit = {
                        is_core_of = ROOT
                        check_variable = { state_population_k > 0 }
                    }
                    set_temp_variable = { temp_remove = ROOT.temp_delta_k }
                    multiply_variable = { temp_remove = state_population_k }
                    divide_variable = { temp_remove = ROOT.max_manpower_k }
                    multiply_variable = { temp_remove = 1000 }
                    round_variable = temp_remove
                    multiply_variable = { temp_remove = -1 }
                    add_manpower = temp_remove
                }
                add_to_variable = { casualties_calculated_deaths_k = temp_delta_k }
            }
        }
    }
}
