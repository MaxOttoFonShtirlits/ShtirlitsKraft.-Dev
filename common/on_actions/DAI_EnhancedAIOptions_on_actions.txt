on_actions = {
	on_startup = {
		effect = {
			every_country = {
				limit = { is_ai = yes }
				DAI_dynamic_construction_monthly_update = yes
			}
			every_country = {
				limit = { is_ai = no }
				country_event = { id = EnhancedAIOptions.1 }
			}
		}
	}

	on_monthly = {
		effect = {
			every_country = {
				limit = { is_ai = yes }
				DAI_dynamic_construction_monthly_update = yes
			}
		}
	}
}
