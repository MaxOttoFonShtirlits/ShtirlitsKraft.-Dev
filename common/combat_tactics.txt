tactic_counterattack = { # counters basic attack, assault
	is_attacker = no
	trigger = {
		is_attacker = no
		phase = no
		OR = {
			skill_advantage > 0
			defense_skill_level > 4
			planning_skill_level > 4
		}
	}
	
	active = yes
	
	base =  {
		factor = 4
		modifier = {
			add = 4
			has_trait = unyielding_defender
		}
	}
	
	picture = counter_battery_fire
	
	countered_by = tactic_elastic_offensive
	
	defender = 0.25
}

tactic_trap = { # Defender springs an ambush — enters complex_ambush phase
	is_attacker = no
	trigger = {
		is_attacker = no
		phase = no
		OR = {
			has_trait = ambusher
			has_trait = trickster
			skill > 4
		}
	}

	active = yes

	base = {
		factor = 1
		modifier = {
			add = 3
			has_trait = ambusher
		}
		modifier = {
			add = 2
			has_trait = trickster
		}
		modifier = {
			add = 1
			skill > 4
		}
	}

	picture = trap

	phase = complex_ambush
	display_phase = complex_ambush

	attacker_movement_speed = 0.10
	defender = 0.00
}
