############################################

FAI_green_air_update = {
	if = {
		limit = { has_war = yes }

		set_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
		every_allied_country = {
			add_to_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
		}
		every_enemy_country = {
			subtract_from_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
		}

		if = {
			limit = {
				NOT = { has_country_flag = FAI_green_air }
				if = {
					OR = {
						check_variable = { FAI_friendly_air > 0 }
						check_variable = { FAI_friendly_air = 0 }
					}
				}
			}
			set_country_flag = FAI_green_air
		}
		else_if = {
			limit = {
				has_country_flag = FAI_green_air
				if = {
					check_variable = { FAI_friendly_air < 0 }
				}
			}
			clr_country_flag = FAI_green_air
		}
	}
	else = { set_country_flag = FAI_green_air }
}

FAI_track_fighter_losses = {
	set_temp_variable = { FAI_previous_fighters_temp = FAI_previous_fighters }
	subtract_from_variable = { FAI_previous_fighters = num_deployed_planes_with_type@small_plane_airframe }
	set_variable = { FAI_monthly_fighters_lost_fraction = FAI_previous_fighters }
	if = {
		limit = { check_variable = { FAI_previous_fighters > 0 } }
		divide_variable = { var = FAI_monthly_fighters_lost_fraction value = FAI_previous_fighters_temp }
	}
	else = {
		set_variable = { FAI_monthly_fighters_lost_fraction = 0 }
	}
	set_variable = { FAI_previous_fighters = num_deployed_planes_with_type@small_plane_airframe }
}

FAI_save_airforce_check = {
	if = {
		limit = { has_war = yes has_capitulated = no }

		### Activate: losing fighters fast or running low
		if = {
			limit = {
				NOT = { has_country_flag = FAI_save_airforce }
				OR = {
					NOT = { has_deployed_air_force_size = { size > 1000 type = fighter } }
					check_variable = { FAI_monthly_fighters_lost_fraction > 0.25 }
				}
			}

			set_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
			for_each_scope_loop = { array = allies
				add_to_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
			}

			every_enemy_country = {
				limit = {
					OR = { is_faction_leader = yes NOT = { is_in_faction = yes } }
				}

				set_temp_variable = { FAI_faction_total = num_deployed_planes_with_type@small_plane_airframe }
				if = { limit = { is_faction_leader = yes }
					every_enemy_country = {
						limit = { is_in_faction_with = PREV NOT = { is_faction_leader = yes } }
						add_to_temp_variable = { FAI_faction_total = num_deployed_planes_with_type@small_plane_airframe }
					}
				}

				set_temp_variable = { FAI_air_diff = FAI_faction_total }
				subtract_from_temp_variable = { FAI_air_diff = FAI_friendly_air }
				set_temp_variable = { FAI_enemy_clamped = FAI_faction_total }
				clamp_temp_variable = { var = FAI_enemy_clamped min = 1 }
				set_temp_variable = { FAI_air_ratio = FAI_friendly_air }
				divide_temp_variable = { FAI_air_ratio = FAI_enemy_clamped }

				if = {
					limit = {
						check_variable = { FAI_air_diff > 800 }
						check_variable = { FAI_air_ratio < 0.25 }
					}
					ROOT = {
						set_country_flag = FAI_save_airforce
						set_country_flag = { flag = FAI_save_airforce_min_duration value = 1 days = 60 }
					}
				}
			}
		}

		### Deactivate: min duration expired
		else_if = {
			limit = {
				has_country_flag = FAI_save_airforce
				NOT = { has_country_flag = FAI_save_airforce_min_duration }
			}

			# Path 1: rebuilt airforce
			if = {
				limit = { has_deployed_air_force_size = { size > 2000 type = fighter } }
				clr_country_flag = FAI_save_airforce
			}

			# Path 2: enemies no longer overwhelming
			else = {
				set_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
				for_each_scope_loop = { array = allies
					add_to_temp_variable = { FAI_friendly_air = num_deployed_planes_with_type@small_plane_airframe }
				}

				set_temp_variable = { FAI_enemies_strong = 0 }
				every_enemy_country = {
					limit = {
						OR = { is_faction_leader = yes NOT = { is_in_faction = yes } }
					}

					set_temp_variable = { FAI_faction_total = num_deployed_planes_with_type@small_plane_airframe }
					if = { limit = { is_faction_leader = yes }
						every_enemy_country = {
							limit = { is_in_faction_with = PREV NOT = { is_faction_leader = yes } }
							add_to_temp_variable = { FAI_faction_total = num_deployed_planes_with_type@small_plane_airframe }
						}
					}

					set_temp_variable = { FAI_air_diff = FAI_faction_total }
					subtract_from_temp_variable = { FAI_air_diff = FAI_friendly_air }
					set_temp_variable = { FAI_enemy_clamped = FAI_faction_total }
					clamp_temp_variable = { var = FAI_enemy_clamped min = 1 }
					set_temp_variable = { FAI_air_ratio = FAI_friendly_air }
					divide_temp_variable = { FAI_air_ratio = FAI_enemy_clamped }

					if = {
						limit = {
							OR = {
								check_variable = { FAI_air_diff > 400 }
								check_variable = { FAI_air_ratio < 0.30 }
							}
						}
						set_temp_variable = { FAI_enemies_strong = 1 }
					}
				}

				if = { limit = { check_variable = { FAI_enemies_strong = 0 } }
					clr_country_flag = FAI_save_airforce
				}
			}
		}
	}
	else = { clr_country_flag = FAI_save_airforce }
}

FAI_equipment_ratio_update = {
	if = {
		limit = { has_war = yes has_capitulated = no }

		# Guard against division by zero (no armies deployed = no crisis)
		if = {
			limit = { check_variable = { num_target_equipment_in_armies_k@infantry_equipment > 0 } }
			set_temp_variable = { FAI_eq_ratio = num_equipment_in_armies_k@infantry_equipment }
			divide_temp_variable = { FAI_eq_ratio = num_target_equipment_in_armies_k@infantry_equipment }
			set_variable = { FAI_fielded_eq_ratio = FAI_eq_ratio }
		}
		else = { set_variable = { FAI_fielded_eq_ratio = 1 } }

		# Activate: armies below 75% equipped (65% for Mass Assault)
		if = {
			limit = {
				NOT = { has_country_flag = FAI_eq_crisis }
				if = {
					limit = { FAI_mass_assault_doctrine = yes }
					check_variable = { FAI_fielded_eq_ratio < 0.65 }
				}
				else = {
					check_variable = { FAI_fielded_eq_ratio < 0.75 }
				}
			}
			set_country_flag = FAI_eq_crisis
		}
		# Deactivate: recovered to 90% (85% for Mass Assault)
		else_if = {
			limit = {
				has_country_flag = FAI_eq_crisis
				if = {
					limit = { FAI_mass_assault_doctrine = yes }
					check_variable = { FAI_fielded_eq_ratio > 0.85 }
				}
				else = {
					check_variable = { FAI_fielded_eq_ratio > 0.9 }
				}
			}
			clr_country_flag = FAI_eq_crisis
		}
	}
	else = { clr_country_flag = FAI_eq_crisis }
}

FAI_manpower_crisis_update = {
	if = {
		limit = { has_war = yes has_capitulated = no }

		if = {
			limit = {
				NOT = { has_country_flag = FAI_manpower_crisis }
				if = {
					limit = { FAI_mass_assault_doctrine = yes }
					manpower_per_military_factory < 300
				}
				else = {
					manpower_per_military_factory < 500
				}
			}
			set_country_flag = FAI_manpower_crisis
		}
		else_if = {
			limit = {
				has_country_flag = FAI_manpower_crisis
				if = {
					limit = { FAI_mass_assault_doctrine = yes }
					manpower_per_military_factory > 700
				}
				else = {
					manpower_per_military_factory > 1000
				}
			}
			clr_country_flag = FAI_manpower_crisis
		}
	}
	else = { clr_country_flag = FAI_manpower_crisis }
}

	FAI_track_fighter_losses = yes
	FAI_save_airforce_check = yes
}

############################################
FAI_weekly_air_update = {
	FAI_green_air_update = yes
	FAI_equipment_ratio_update = yes
	FAI_manpower_crisis_update = yes
}

FAI_operational_pause_update = {

	# When war starts, stagger initialization across countries (1-in-61 chance daily)
	if = {
		limit = {
			has_war = yes
			NOT = { has_country_flag = FAI_op_pause_initialized }
		}
		random_list = {
			1 = { set_country_flag = FAI_op_pause_initialized }
			60 = {}
		}
	}
	else_if = {
		limit = {
			has_war = no
			has_country_flag = FAI_op_pause_initialized
		}
		clr_country_flag = FAI_op_pause_initialized
	}

	# Start new cycle when initialized and no active timer remains
	if = {
		limit = {
			has_country_flag = FAI_op_pause_initialized
			NOT = {
				has_country_flag = FAI_op_pause_timer
				has_country_flag = FAI_op_resume_timer
			}
		}

		random_list = {
			10 = {
				set_country_flag = { flag = FAI_op_pause_timer value = 1 days = 7 }
				set_country_flag = { flag = FAI_op_resume_timer value = 1 days = 67 }
			}
			10 = {
				set_country_flag = { flag = FAI_op_pause_timer value = 1 days = 9 }
				set_country_flag = { flag = FAI_op_resume_timer value = 1 days = 79 }
			}
			10 = {
				set_country_flag = { flag = FAI_op_pause_timer value = 1 days = 11 }
				set_country_flag = { flag = FAI_op_resume_timer value = 1 days = 91 }
			}
			10 = {
				set_country_flag = { flag = FAI_op_pause_timer value = 1 days = 14 }
				set_country_flag = { flag = FAI_op_resume_timer value = 1 days = 114 }
			}
		}
	}
}
